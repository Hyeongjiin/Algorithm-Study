from collections import deque
def solution(n, computers):
    answer = 0
    visited = [False] * n
    for i in range(n):
        if visited[i] == False:
            answer += 1
            bfs(i, visited, computers)
    return answer

def bfs(start, visited, computers):
    q = deque()
    q.append(start)
    visited[start] = True
    while q:
        v = q.popleft()
        for i in range(len(computers)):
            if i == v:
                continue
            if computers[v][i] == 1 and visited[i] == False:
                visited[i] = True
                q.append(i)
