from collections import deque
def solution(number, k):
    answer = ''
    number = deque(list(number))
    result = []
    result.append(number.popleft())
    while k > 0:
        if result[-1] >= number[0]:
            result.append(number.popleft())
        elif result[-1] < number[0]:
            result.pop()
            if len(result) == 0:
                result.append(number.popleft())
            k -= 1
        if len(number) == 0:
            break
    if k > 0:
        for i in range(k):
            result.pop()
    for i in result:
        answer += i
    for i in number:
        answer += i
    return answer
