from collections import deque
def solution(bridge_length, weight, truck_weights):
    answer = 0
    total_truck = len(truck_weights)
    truck_left = total_truck
    truck_on_road = 0
    truck_done = 0
    truck_wait = deque(truck_weights)
    truck_move = deque()
    current_weight = 0
    time = 0
    while truck_done < total_truck:
        if truck_left > 0:
            if current_weight + truck_wait[0] <= weight:
                current_weight += truck_wait[0]
                truck_on_road += 1
                truck_left -= 1
                truck_move.append((0, truck_wait[0]))
                truck_wait.popleft()
        for i in range(truck_on_road):
            truck_move[i] = (truck_move[i][0] + 1, truck_move[i][1])
        time += 1
        if truck_on_road > 0:
            if truck_move[0][0] == bridge_length:
                current_weight -= truck_move[0][1]
                truck_move.popleft()
                truck_on_road -= 1
                truck_done += 1
    answer = time + 1
    return answer
