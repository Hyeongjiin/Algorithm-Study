def solution(n):
    answer = 0
    if n == 1:
        answer = 1
        return answer
    elif n == 2:
        answer = 2
        return answer
    dp = [0] * (n + 1)
    dp[1] = 1
    dp[2] = 2
    for i in range(3, n + 1):
        dp[i] = (dp[i - 1] + dp[i - 2]) % 1000000007
    answer = dp[n] % 1000000007
    return answer
