def solution(n, m):
    mini = 0
    maxi = 0
    if n <= m:
        mini = n
        maxi = m
    else: 
        mini = m
        maxi = n
    g = 0
    c = 0
    for i in range(1, mini + 1):
        if n % i == 0 and m % i == 0:
            g = i
    c = mini * maxi // g
    answer = [g, c]
    return answer
