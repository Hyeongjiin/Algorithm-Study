def solution(sizes):
    answer = 0
    for i in sizes:
        if i[0] < i[1]:
            i[0], i[1] = i[1], i[0]
    maxi1 = 0
    maxi2 = 0
    for i in sizes:
        if maxi1 < i[0]:
            maxi1 = i[0]
        if maxi2 < i[1]:
            maxi2 = i[1]
    answer = maxi1 * maxi2
    return answer
