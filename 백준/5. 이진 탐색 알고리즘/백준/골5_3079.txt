import sys
input = sys.stdin.readline

def b_search(data, M, start, end):
    result = 0
    while start <= end:
        count = 0
        mid = (start + end) // 2
        for i in data:
            count += mid // i
        if count >= M:
            result = mid
            end = mid - 1
        else:
            start = mid + 1
    return result
        

N, M = map(int, input().split())
data = []
for i in range(N):
    data.append(int(input()))
end = max(data) * M
result = b_search(data, M, 1, end)
print(result)
