import sys

def b_search(data, M, start, end):
    minimum = 0
    while start <= end:
        mid = (start + end) // 2
        money = mid
        count = 1
        for i in data:
            if money < i:
                money = mid - i
                count += 1
            else:
                money -= i
        if count > M:
            start = mid + 1
        else:
            end = mid - 1
            minimum = mid
    return minimum

input = sys.stdin.readline

N, M = map(int, input().split())
data = []
for i in range(N):
    data.append(int(input()))
start = 0
sum = 0
start = 0
for i in data:
    sum += i
    if i > start:
        start = i

result = b_search(data, M, start, sum)
print(result)
