import sys
input = sys.stdin.readline
T = int(input())
for i in range(T):
    N = int(input())
    if N >= 2:
        Zero = [0] * (N + 1)
        Zero[0] = 1
        One = [0] * (N + 1)
        One[1] = 1
        for i in range(2, N + 1):
            Zero[i] = Zero[i - 1] + Zero[i - 2]
            One[i] = One[i - 1] + One[i - 2]
        print(Zero[N], One[N])
    else:
        if N == 0:
            print(1, 0)
        else: 
            print(0, 1)
