import sys
input = sys.stdin.readline
N = int(input())
if N >= 2:
    Two = [0] * (N - 1)
    Two[0] = 1
    One = [0] * (N - 1)
    for i in range(1, N - 1):
        Two[i] = (Two[i - 1] + One[i - 1]) % 10
        One[i] = Two[i - 1] % 10
    result = (Two[N - 2] * 2 + One[N - 2]) % 10
    print(result)
else:
    print(1)
