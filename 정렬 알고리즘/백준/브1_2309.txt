import sys
input = sys.stdin.readline
h = []
for i in range(9):
    h.append(int(input()))
h.sort()
total = sum(h)
f = 0
s = 0
for i in range(8):
    for j in range(i + 1, 9):
            if total - (h[i] + h[j]) == 100:
                f = i
                s = j

for i in range(9):
    if i == f or i == s:
        continue
    else:
        print(h[i])
