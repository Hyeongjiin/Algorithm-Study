import math

M, N = map(int, input().split())
data = [True for _ in range(N + 1)]
data[1] = 0

for i in range(2, int(math.sqrt(N)) + 1):
    if data[i]:
        j = 2
        while i * j <= N:
            data[i * j] = False
            j += 1

for i in range(M, N + 1):
    if data[i]:
        print(i)
