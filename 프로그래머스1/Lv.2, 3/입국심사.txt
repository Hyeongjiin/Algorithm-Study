def solution(n, times):
    answer = 0
    length = len(times)
    start = 1
    end = max(times) * n
    while start <= end:
        result = 0
        mid = (start + end) // 2
        for i in times:
            result += mid // i
        if result < n:
            start = mid + 1
        else:
            end = mid - 1
            answer = mid
    return answer
