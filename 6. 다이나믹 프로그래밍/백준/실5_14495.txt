import sys

def fibo(N):
    for i in range(3, N):
        data[i] = data[i - 1] + data[i - 3]

input = sys.stdin.readline
N = int(input())
data = [0] * N
data[0] = 1
if N >= 2:
    data[1] = 1
if N >= 3:
    data[2] = 1

if N == 1 or N == 2 or N == 3:
    print(1)
else:
    fibo(N)
    print(data[N - 1])
