N = int(input())
data = list(map(int, input().split()))
count_max = 0
count_min = 0
max = 0

for i in range(1, len(data)):
    if data[i] > data[i - 1]:
        count_max += 1
        count_min = 0
    elif data[i] < data[i - 1]:
        count_max = 0
        count_min += 1
    else:
        count_max += 1
        count_min += 1
    if count_max > max:
        max = count_max
    if count_min > max:
        max = count_min

print(max + 1)
