import sys

def b_search(data, N, start, end):
    result = 0
    while start <= end:
        count = 0
        mid = (start + end) // 2
        for i in data:
            count += i // mid
        if count < N:
            end = mid - 1
        else:
            start = mid + 1
            result = mid
    return result

input = sys.stdin.readline

K, N = map(int, input().split())
data = []
for i in range(K):
    data.append(int(input()))
start = 0
end = 2 * max(data)
length = b_search(data, N, start, end)
print(length)
