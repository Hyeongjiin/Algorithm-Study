data = input()
count_M = 0
length = len(data)
max = ""
min = ""

for i in range(length):
    if data[i] == "M":
        count_M += 1
    else:
        if (count_M == 0):
            max += "5"
            min += "5"
        else:
            max += str(10 ** count_M * 5)
            min += str(10 ** (count_M - 1))
            min += "5"
        count_M = 0

if (count_M > 0):
    max += "1" * count_M
    min += str(10 ** (count_M - 1))

print(max)
print(min)
