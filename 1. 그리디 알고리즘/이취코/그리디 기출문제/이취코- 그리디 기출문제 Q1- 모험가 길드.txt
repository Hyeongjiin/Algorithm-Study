N = int(input())
data = list(map(int, input().split()))
data.sort()
group = 0
count = 0

for i in range(N):
    if (i + 1) - count >= data[i]:
        group += 1
        count += data[i]

print(group)

N = int(input())
data = list(map(int, input().split()))
data.sort()
group = 0
count = 0

for i in range(N):
    count += 1
    if data[i] == count:
        group += 1
        count = 0

print(group)
