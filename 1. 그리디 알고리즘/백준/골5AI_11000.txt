import heapq
import sys
input = sys.stdin.readline

N = int(input())
data = []
for i in range(N):
    S, T = map(int, input().split())
    data.append((S, T))
data = sorted(data, key = lambda x: (x[0], x[1]))
q = []
heapq.heappush(q, data[0][1])
for i in range(1, N):
    if data[i][0]< q[0]:
        heapq.heappush(q, data[i][1])
    else:
        heapq.heappop(q)
        heapq.heappush(q, data[i][1])
print(len(q))
