import sys
input = sys.stdin.readline
N, L = map(int, input().split())
data = list(map(int, input().split()))
data.sort()
current = 0
progress = 0
count = 0

while N > 0:
    if data[current + progress] <= data[current] + L - 1:
        N -= 1
        progress += 1
    else:
        count += 1
        current += progress
        progress = 0

if progress > 0:
    count += 1

print(count)
