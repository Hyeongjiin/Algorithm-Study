T = int(input())
count = [0, 0, 0]
flag = 0

if T % 10 != 0:
    flag = 1
    print(-1)

while T >= 10 and flag == 0:
    if T >= 300:
        T -= 300
        count[0] += 1
    elif T >= 60:
        T -= 60
        count[1] += 1
    elif T >= 10:
        T -= 10
        count[2] += 1
      
if flag == 0:
    print(count[0], count[1], count[2])
