import sys
input = sys.stdin.readline
N = int(input())
data = [int(input()) for _ in range(N)]

if N == 1:
    print(1)
else:
    if data[0] >= data[1]:
        print(1)
    for i in range(1, N - 1):
        if data[i - 1] <= data[i] >= data[i + 1]:
            print(i + 1)
    if data[N - 1] >= data[N - 2]:
        print(N)
