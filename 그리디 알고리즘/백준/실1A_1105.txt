L, R = list(input().split())

count_L = L.count("8")
count_R = R.count("8")
mini = 0

if count_L == 0 or count_R == 0:
    print(0)
elif len(L) != len(R):
    print(0)
elif len(L) == len(R):
    for i in range(len(L)):
        if L[i] == R[i]:
            if L[i] == "8":
                mini += 1
        else:
            break
    print(mini)
