import sys

def b_search(data, goal, start, end):
    result = 0
    while start <= end:
        count = 1
        mid = (start + end) // 2
        stand = data[0]
        for i in range(1, len(data)):
            if data[i] - stand >= mid:
                stand = data[i]
                count += 1
                if count == goal:
                    break
        if count == goal:
            result = mid
            start = mid + 1
        else:
            end = mid - 1
    return result
        
input = sys.stdin.readline

N, C = map(int, input().split())
data = []
for i in range(N):
    data.append(int(input()))
data.sort()
start = 1
end = data[-1] * 2
result = b_search(data, C, start, end)
print(result)
