import sys
input = sys.stdin.readline
N = int(input())
w = [0] * N
w[0] = 1
h = [0] * N
h[0] = 1
for i in range(1, N):
    if i % 2 == 1:
        h[i] = h[i - 1] + w[i - 1]
        w[i] = w[i - 1]
    else:
        w[i] = w[i - 1] + h[i - 1]
        h[i] = h[i - 1]
result = (h[N - 1] + w[N - 1]) * 2
print(result)
